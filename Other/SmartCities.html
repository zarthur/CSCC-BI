def zip_boundary(points):
    min_x = 180
    max_x = -180
    min_y = 90
    max_y = -90
    for p in points:
        if p.x < min_x:
            min_x = p.x
        if p.x > max_x:
            max_x = p.x
        if p.y < min_y:
            min_y = p.y
        if p.y > max_y:
            max_y = p.y
    return min_x, min_y, max_x, max_y

def greatest_minimum_distance(addrs, stops):
    number_of_addresses = len(addrs)
    greatest_minimum_distance = 0
    for index, address_point in enumerate(addrs.geometry):
        minimum_stop_distance = distance(address_point, stops.iloc[0].geometry)
        for stop_point in stops.geometry:
            stop_distance = distance(address_point, stop_point)
            if stop_distance < minimum_stop_distance:
                minimum_stop_distance = stop_distance

        if minimum_stop_distance > greatest_minimum_distance:
            greatest_minimum_distance = minimum_stop_distance
    return greatest_minimum_distance

for i, zipcode in enumerate(unique_zipcodes):
    addresses_zip = addresses[addresses.ZIP == zipcode]
    if len(addresses_zip) > 1000:
        addresses_zip = addresses_zip.sample(n=1000)
    if len(addresses_zip) < 100:
        print("Zip with few addresses: " + str(zipcode))

    df_six = df.copy()
